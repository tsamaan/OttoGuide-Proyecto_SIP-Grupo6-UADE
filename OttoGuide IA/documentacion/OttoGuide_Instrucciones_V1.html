⌨️ Comandos básicos de operación
bashReferencia rápida
# ── STACK ───────────────────────────────────────── docker compose up -d # levantar todo docker compose down # bajar todo docker compose restart # reiniciar todo docker compose ps # ver estado de contenedores # ── LOGS ────────────────────────────────────────── docker compose logs -f # logs de todos en tiempo real docker compose logs -f llm # logs solo del LLM docker compose logs -f core # logs del orquestador # ── CONTENEDOR INDIVIDUAL ───────────────────────── docker exec -it ottoguide-llm bash # entrar al contenedor LLM docker exec -it ottoguide-core bash # entrar al core # ── REBUILD (cuando cambia código) ──────────────── docker compose up -d --build core # rebuild solo core docker compose up -d --build # rebuild todo

⚡ Auto-start con systemd 🤖 Robot
inideploy/ottoguide.service
[Unit] Description=OttoGuide AI Stack After=docker.service network-online.target Wants=network-online.target Requires=docker.service [Service] Type=simple WorkingDirectory=/opt/ottoguide ExecStartPre=docker compose pull ExecStart=docker compose up ExecStop=docker compose down Restart=on-failure RestartSec=10 [Install] WantedBy=multi-user.target
bashInstalar el servicio en el robot (Fase 5)
# Copiar el servicio y habilitarlo (en el robot) sudo cp deploy/ottoguide.service /etc/systemd/system/ sudo systemctl daemon-reload sudo systemctl enable ottoguide sudo systemctl start ottoguide # Verificar estado sudo systemctl status ottoguide
✓ Con este servicio configurado, al encender el robot el stack de IA se levanta automáticamente en segundo plano. Si algo falla, systemd lo reinicia solo después de 10 segundos.
